struct tChorizo {
    int disponible;
    int cBollo;
    int mBollo;
    int precio;
};

int maxBeneficio(vector<tChorizo>& chorizos, int masa) {
    Matriz<int> matriz(chorizos.size() + 1, masa + 1, 0);

    for (int i = 1; i <= chorizos.size(); i++) {
        vector<int> disp(masa + 1, chorizos[i - 1].disponible);
        /*
        for (int j = 0; j <= masa; j++) {
            cout << disp[j] << " ";
        }
        cout << "\n";*/
        for (int j = 1; j <= masa; j++) {
            if (chorizos[i - 1].mBollo > j) {
                matriz[i][j] = matriz[i - 1][j];
            } else if (disp[j - chorizos[i - 1].mBollo] < chorizos[i - 1].cBollo) {
                matriz[i][j] = max(matriz[i - 1][j], matriz[i][j - chorizos[i - 1].mBollo]);
                if (matriz[i][j] != matriz[i - 1][j]) {
                    disp[j] = disp[j - chorizos[i - 1].mBollo];
                }
                // cout << matriz[i][j] << " ";
            } else {
                matriz[i][j] = max(matriz[i - 1][j], matriz[i][j - chorizos[i - 1].mBollo] + chorizos[i - 1].precio);
                if (matriz[i][j] != matriz[i - 1][j]) {
                    disp[j] = disp[j - chorizos[i - 1].mBollo] - chorizos[i - 1].cBollo;
                }
                if (matriz[i - 1][j] != matriz[i - 1][j - chorizos[i - 1].mBollo]) {
                    matriz[i][j] += matriz[i - 1][j] - matriz[i - 1][j - chorizos[i - 1].mBollo];
                }
                // cout << "Hola\n";

                //  cout << matriz[i][j] << endl;
            }
        }
        // cout << endl;
        /*
        for (int j = 0; j <= masa; j++) {
            cout << disp[j] << " ";
        }
        cout << "\n";*/
    }
    /*
    for (int i = 0; i <= masa; i++) {
        cout << " " << i;
    }
    cout << "\n";
    for (int i = 0; i <= chorizos.size(); i++) {
        for (int j = 0; j <= masa; j++) {
            cout << " " << matriz[i][j];
        }
        cout << "\n";
    }
*/
    return matriz[chorizos.size()][masa];
}

bool resuelveCaso() {
    // leer los datos de la entrada

    vector<tChorizo> chorizos(N);

}
