struct tPeli {
    int ini;
    int fin;
    int sol;
};

bool operator<(tPeli p1, tPeli p2) {
    return p1.fin < p2.fin;
}

// peli(i) = max(peli(i+1), peli(i+1) + duracion)

int maxPelis(vector<tPeli>& pelis, int i, int fin) {
    if (i >= pelis.size()) {
        return 0;
    }
    if (fin + 10 > pelis[i].ini) {
        // pelis[i].sol = pelis[i + 1].sol;
        pelis[i].sol = maxPelis(pelis, i + 1, fin);
        return pelis[i].sol;
    } else {
        int duracion = pelis[i].fin - pelis[i].ini;
        pelis[i].sol = max(maxPelis(pelis, i + 1, pelis[i].fin) + duracion, pelis[i + 1].sol);
        cout << "pelis[ " << i << " ] = " << pelis[i].sol << endl;
        return pelis[i].sol;
    }
}
