struct tHora {
    int hora = 0;
    int min = 0;
    int distancia;
};

int distMin(tHora a, tHora b) {  // es esto porque es un reloj circular. hay que tener en cuenta que 23:58 esta a dos minutos de 00:00
    int d = abs((a.hora * 60 + a.min) - (b.hora * 60 + b.min));
    return min(d, 1440 - d);  // reloj circular
}
tHora bfs(tHora ini, tHora fin, const vector<tHora>& opc) {
    tHora mejor = ini;
    ini.distancia = 0;
    bool visitado[24][60] = {false};
    visitado[ini.hora][ini.min] = true;
    queue<tHora> cola;
    cola.push(ini);
    while (!cola.empty()) {
        tHora v = cola.front();
        cola.pop();

        for (int i = 0; i < 2; i++) {
            int auxH, auxM;
            auxH = (opc[i].hora + v.hora) % 24;
            auxM = (opc[i].min + v.min) % 60;

            if (!visitado[auxH][auxM]) {
                visitado[auxH][auxM] = true;
                tHora aux{auxH, auxM, v.distancia + 1};
                cola.push(aux);
                if (fin.hora == aux.hora && aux.min == fin.min) {
                    return aux;
                }

                if (distMin(fin, mejor) > distMin(fin, aux)) {
                    mejor = aux;
                } else if (distMin(fin, mejor) == distMin(fin, aux)) {
                    if (aux.hora < mejor.hora) {
                        mejor = aux;
                    } else if (aux.hora == mejor.hora && aux.min < mejor.min) {
                        mejor = aux;
                    }
                }
                /*
